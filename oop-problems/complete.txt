游릮 3. Functions
游댳 Basic Function

游댳 Default & Variadic Parameters

游댳 Static Variables

    static $count = 0;

    return $count;

游릭 10. Object-Oriented Programming (OOP)
游댳 Class and Object
class Car {
    public $brand;
    public function start() {
        echo "$this->brand is starting!";
    }
}
$car = new Car();
$car->brand = "Toyota";
$car->start();
游댳 Constructor & Destructor
class User {
    public function __construct($name) {
        echo "User $name created.<br>";
    }
    public function __destruct() {
        echo "User destroyed.<br>";
    }
}
$user = new User("Kamil");
游댳 Access Modifiers
class Bank {
    private $balance = 0;
    public function deposit($amount) {
        $this->balance += $amount;
    }
    public function getBalance() {
        return $this->balance;
    }
}
$acc = new Bank();
$acc->deposit(100);
echo $acc->getBalance();
游댳 Inheritance
class Vehicle {
    public function start() {
        echo "Vehicle started.";
    }
}
class Car extends Vehicle {
    public function honk() {
        echo "Beep!";
    }
}
$car = new Car();
$car->start();
$car->honk();
游댳 Method Overriding

游댳 Static Properties & Methods
class Math {
    public static $pi = 3.14;
    public static function square($n) {
        return $n * $n;
    }
}

echo Math::square(5);
游댳 Abstract Class
abstract class Shape {
    abstract public function area();
}
class Circle extends Shape {
    private $r;
    public function __construct($r) { $this->r = $r; }
    public function area() { return pi() * $this->r ** 2; }
}
echo (new Circle(3))->area();
游댳 Interface
interface Logger {
    public function log($msg);
}
class FileLogger implements Logger {
    public function log($msg) {
        echo "Log: $msg";
    }
}
(new FileLogger())->log("Success");
游댳 Trait
trait Loggable {
    public function log($msg) {
        echo "[LOG]: $msg";
    }
}
class Product {
    use Loggable;
}
$p = new Product();
$p->log("Product created");
游댳 Magic Methods
class Demo {
    private $data = [];
    public function __set($name, $value) {
        $this->data[$name] = $value;
    }
    public function __get($name) {
        return $this->data[$name] ?? null;
    }
    public function __toString() {
        return json_encode($this->data);
    }
}
$d = new Demo();
$d->name = "Kamil";
echo $d;
